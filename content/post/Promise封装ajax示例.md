/*使用Promise封装ajax*/
function getList(url, data) {
	return new Promise(function (resolve, reject) {
		$.ajax({
			url: HanderServiceUrl+url,
			type: "POST",
			dataType: 'json',
			data: data,
			success: function (json) {
				resolve(json)
			},
			error: function (err) {
				reject(json)
			}
		})
	})
}
